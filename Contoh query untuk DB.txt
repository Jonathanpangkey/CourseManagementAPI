Ini merupakan contoh kode yang bisa diterapkan 
untuk membuat rancangan database yang diberikan 
dengan menggunakan mongoDB

// Buat database MongoDB baru
use myDatabase

// Buat koleksi admins
db.createCollection("admins")
// Masukkan data dengan attribut yang sesuai dengan soal
db.admins.insertOne({
  "id": 1,
  "name": "admin",
  "email": "admin@contoh.com",
  "password": "password123"
})

// Buat koleksi course_categories
db.createCollection("course_categories")
// Masukkan data dengan attribut yang sesuai dengan soal
db.course_categories.insertOne({
  "id": 1,
  "name": "Pemrograman"
})

// Buat koleksi courses
db.createCollection("courses")
// Masukkan data dengan attribut yang sesuai dengan soal
db.courses.insertOne({
  "id": 1,
  "title": "Pengenalan Python",
  "course_categoryId": 1
})

// Buat koleksi users
db.createCollection("users")
// Masukkan data dengan attribut yang sesuai dengan soal
db.users.insertOne({
  "id": 1,
  "name": "John Doe",
  "email": "john@example.com",
  "password": "password123"
})

// Buat koleksi user_course
db.createCollection("user_course")
// Masukkan data dengan attribut yang sesuai dengan soal
db.user_course.insertOne({
  "id": 1,
  "user_id": 1,
  "course_id": 1
})
